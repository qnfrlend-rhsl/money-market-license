    function generateLotto() {
      let numbers = [];

      while (numbers.length < 6) {
        let num = Math.floor(Math.random() * 45) + 1;
        if (!numbers.includes(num)) {
          numbers.push(num);
        }
      }

      numbers.sort((a, b) => a - b);

      function getBallColor(num) {
        if (num <= 10) return "radial-gradient(circle at 30% 30%, #fff, #ffd633)";
        if (num <= 20) return "radial-gradient(circle at 30% 30%, #fff, #4d79ff)";
        if (num <= 30) return "radial-gradient(circle at 30% 30%, #fff, #ff4d4d)";
        if (num <= 40) return "radial-gradient(circle at 30% 30%, #fff, #333333)";
        return "radial-gradient(circle at 30% 30%, #fff, #66f308)";
      }

      const resultDiv = document.getElementById("result");
      resultDiv.innerHTML = "";

      const delayStep = 900;

      numbers.forEach((n, i) => {
        setTimeout(() => {
          const span = document.createElement("div");
          span.textContent = n;
          span.classList.add("ball");
          span.style.background = getBallColor(n);
          resultDiv.appendChild(span);
        }, i * delayStep);
      });

      setTimeout(() => {
        const emoji = document.createElement("div");
        emoji.innerHTML = "🎉";
        emoji.classList.add("celebrate");
        resultDiv.appendChild(emoji);
      }, numbers.length * delayStep + 200);
    }
  </script>
